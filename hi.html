startBtn.addEventListener('click', async () => { try { const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); // 아이폰/안드로이드 호환성을 위해 지원하는 형식을 체크합니다. const mimeType = MediaRecorder.isTypeSupported('audio/webm') ? 'audio/webm' : 'audio/mp4'; // 아이폰은 mp4 기반 형식을 더 선호할 수 있습니다. mediaRecorder = new MediaRecorder(stream, { mimeType }); audioChunks = []; mediaRecorder.ondataavailable = event => { if (event.data.size > 0) audioChunks.push(event.data); }; mediaRecorder.onstop = () => { const audioBlob = new Blob(audioChunks, { type: mimeType }); const audioUrl = URL.createObjectURL(audioBlob); audioPlayback.src = audioUrl; audioPlayback.classList.remove('hidden'); downloadLink.href = audioUrl; // 확장자도 유동적으로 변경 downloadLink.download = `speaking_practice.${mimeType.split('/')[1]}`; downloadLink.classList.remove('hidden'); stream.getTracks().forEach(track => track.stop()); }; mediaRecorder.start(); startBtn.disabled = true; stopBtn.disabled = false; recordingStatus.classList.remove('hidden'); audioPlayback.classList.add('hidden'); downloadLink.classList.add('hidden'); } catch (err) { alert('마이크 접근 권한이 없거나 지원하지 않는 브라우저입니다.\n반드시 Safari나 정식 Chrome 앱에서 웹 주소로 접속해주세요.'); console.error(err); } });
